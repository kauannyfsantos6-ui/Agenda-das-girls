document.addEventListener('DOMContentLoaded', function() { // Sobrescreve o submit do formulário para redirecionar após agendar e não exigir login document.getElementById('form-agendamento').addEventListener('submit', function(e) { e.preventDefault(); // Verificar se um horário foi selecionado const horarioSelecionado = document.querySelector('#horarios-container .horario-btn.selected'); if (!horarioSelecionado) { alert('Por favor, selecione um horário disponível.'); return; } const user = getCurrentUser(); if (!user || user.role !== 'cliente') { alert('Você precisa entrar como cliente para confirmar o agendamento.'); window.location.href='login.html'; return; } // Obter dados do formulário const nome = document.getElementById('nome').value; const telefone = document.getElementById('telefone').value; const email = document.getElementById('email').value; const servico = document.getElementById('servico').options[document.getElementById('servico').selectedIndex].text; const data = document.getElementById('data').value; const horario = horarioSelecionado.textContent; const stylistEmail = document.getElementById('cabeleireiro').value; const stylistName = document.getElementById('cabeleireiro').selectedOptions[0] ? document.getElementById('cabeleireiro').selectedOptions[0].text : ''; const client = getCurrentUser(); // salvar appointment const appt = { id: Date.now(), clientEmail: client.email, clientName: client.name, stylistEmail: stylistEmail, stylistName: stylistName, service: servico, date: data, time: horario, telefone: telefone }; saveAppointment(appt); localStorage.setItem('lastAppointment', JSON.stringify(appt)); // Salva o último agendamento para confirmação // Redirecionar para confirmacao.html window.location.href = 'confirmacao.html'; }, true); });
